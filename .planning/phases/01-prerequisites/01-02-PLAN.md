Create `app/lib/trendingCache.ts` (D-07 — new module wrapping fetchTrending; do NOT modify fetchTrending).

```typescript
import { fetchTrending } from "./github";
import type { Repo } from "./types";

const TRENDING_TTL_MS = 5 * 60 * 1000; // 5 min, same as the repos route today (D-08)

type Params = { topic?: string; query?: string; since?: string; page?: number; perPage?: number };

const cache = new Map<string, { at: number; data: Repo[] }>();
const inflight = new Map<string, Promise<Repo[]>>();

function keyOf(p: Params): string {
  const topic = (p.topic ?? "").trim().toLowerCase();   // lowercased topic (D-08)
  const query = (p.query ?? "").trim();
  const since = (p.since ?? "").slice(0, 10);
  const page = p.page ?? 1;
  const perPage = p.perPage ?? 30;
  return `${topic}|${query}|${since}|${page}|${perPage}`;
}

export async function fetchTrendingCached(p: Params = {}): Promise<Repo[]> {
  const key = keyOf(p);

  const hit = cache.get(key);
  if (hit && Date.now() - hit.at < TRENDING_TTL_MS) return hit.data;   // serve within TTL

  const pending = inflight.get(key);
  if (pending) return pending;                                          // coalesce concurrent (D-09)

  const promise = (async () => {
    const data = await fetchTrending(p);
    if (data.length > 0) cache.set(key, { at: Date.now(), data });      // cache non-empty only (D-10)
    return data;
  })().finally(() => {
    inflight.delete(key);
  });

  inflight.set(key, promise);
  return promise;
}
```
