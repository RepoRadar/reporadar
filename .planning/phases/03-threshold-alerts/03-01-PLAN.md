CREATE INDEX IF NOT EXISTS idx_subs_verify_token ON subscriptions(verify_token);
CREATE INDEX IF NOT EXISTS idx_subs_unsub_token ON subscriptions(unsub_token);

CREATE TABLE IF NOT EXISTS repo_snapshots (
  term TEXT NOT NULL,
  full_name TEXT NOT NULL,
  stars INTEGER NOT NULL,
  captured_at TEXT NOT NULL,
  PRIMARY KEY (term, full_name, captured_at)

Export typed functions (signatures the downstream plans depend on — keep these EXACT):
```typescript
export type SubscriptionRow = {
  id: string; email: string; kind: "topic" | "query"; term: string;
  metric: "stars_pct" | "stars_abs" | "velocity"; threshold: number;
  window_days: number; digest: string | null; created_at: string;
  verified_at: string | null; last_notified_at: string | null;
  verify_token: string; unsub_token: string;
};
export async function createSubscription(db: D1Database, sub: Omit<SubscriptionRow, "verified_at" | "last_notified_at" | "digest"> & { digest?: string | null }): Promise<void>;
export async function getSubscriptionByVerifyToken(db: D1Database, token: string): Promise<SubscriptionRow | null>;
export async function verifySubscription(db: D1Database, token: string): Promise<boolean>; // returns true if a row was updated
export async function unsubscribeByToken(db: D1Database, token: string): Promise<boolean>;
export async function listSubscriptionsByEmail(db: D1Database, email: string): Promise<SubscriptionRow[]>;
export async function listDistinctTerms(db: D1Database): Promise<{ term: string; kind: "topic" | "query" }[]>; // SELECT DISTINCT term, kind FROM subscriptions WHERE verified_at IS NOT NULL
export async function listVerifiedSubsForTerm(db: D1Database, term: string): Promise<SubscriptionRow[]>; // WHERE term=? AND verified_at IS NOT NULL
export async function getLatestSnapshot(db: D1Database, term: string): Promise<Map<string, number>>; // full_name -> stars at the most recent captured_at for this term
export async function writeSnapshots(db: D1Database, term: string, repos: { fullName: string; stars: number }[], capturedAt: string): Promise<void>; // use db.batch() for multi-row (RESEARCH Pitfall 4)
export async function setLastNotified(db: D1Database, id: string, isoTime: string): Promise<void>;
```
For `getLatestSnapshot`: select the MAX(captured_at) for the term, then the rows at that captured_at, return as a `Map<full_name, stars>`. For `writeSnapshots`: use `db.batch([...prepared statements])` rather than N sequential `.run()` (RESEARCH Pitfall 4). `verifySubscription`/`unsubscribeByToken` return whether a row matched (check `meta.changes`), so routes can give a generic response without leaking token validity (threat: email enumeration / token probing).
