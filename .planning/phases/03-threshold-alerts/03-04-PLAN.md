```typescript
export async function runAlertSweep(env: { DB: D1Database }): Promise<{ sent: number; scanned: number }> {
  const terms = await listDistinctTerms(env.DB);          // dedupe at SQL level
  for (const t of terms) {
    const repos = await fetchTrendingCached(t.kind === "topic" ? { topic: t.term } : { query: t.term });
    const prior = await getLatestSnapshot(env.DB, t.term);
    for (const sub of await listVerifiedSubsForTerm(env.DB, t.term)) {
      for (const c of detectCrossings(sub, repos, prior)) {
        if (alreadyNotified(sub, c)) continue;            // last_notified_at + crossing identity
        await sendAlertEmail(sub, c);                     // sendEmail(buildAlertEmail(...))
        await setLastNotified(env.DB, sub.id, new Date().toISOString());
      }
    }
    await writeSnapshots(env.DB, t.term, repos, capturedAt); // AFTER detection (Pitfall 2)
  }
}
```
